## Install

```bash
npm install --save bro-team-ui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'bro-team-ui'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
